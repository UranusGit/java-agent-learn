# 29 - 成本治理与计费:预算、配额、账单、商业化

> 本章目标:把 10 章的 token 上报升级为完整的成本治理 + 商业化计费体系。
> 完成后:租户 / 用户 / 项目的多级预算、模型分级路由、月度账单、信用卡 / 发票 / 退款、实时余额、超额降级。
>
> **关联章节**:
> - 10 章 UsageReporter 上报 token(本章的基础);
> - 14 章模型 fallback chain(本章按成本路由);
> - 18 章 CircuitBreaker(本章按预算硬熔断);
> - 28 章限流(本章是限流之上的"按钱"治理);
> - 31 章多租户(本章计费归属到 tenant + project)。

## 0. 为什么必须做

10 章只上报用量,生产场景会遇到:

| 场景 | 后果 |
|------|------|
| 客户跑爆 Anthropic 账单 $10k | 拒付,你赔 |
| 用户误用 Claude Opus 跑 100 万 token | 一夜 $50 没了 |
| 内部测试消耗公司全部预算 | 线上用户无服务 |
| 客户要求发票 / 对公付款 | 没法做商务 |
| 客户要求按 project 拆账 | 财务混乱 |
| 不同模型价格差 100x | 没引导用户用便宜模型 |

商业化的核心:让用户**知道自己花了多少** + **超了能自动停**。

```java
// 本代码仅作学习材料参考
public record ModelUsage(
    String model,
    long inputTokens,
    long outputTokens,
    long cachedInputTokens,
    Instant at
) {
    public BigDecimal costUsd(PriceTable prices) {
        Price p = prices.forModel(model);
        BigDecimal in = p.inputPerMillion().multiply(BigDecimal.valueOf(inputTokens))
            .divide(BigDecimal.valueOf(1_000_000), 6, RoundingMode.HALF_UP);
        BigDecimal out = p.outputPerMillion().multiply(BigDecimal.valueOf(outputTokens))
            .divide(BigDecimal.valueOf(1_000_000), 6, RoundingMode.HALF_UP);
        BigDecimal cached = p.cachedPerMillion().multiply(BigDecimal.valueOf(cachedInputTokens))
            .divide(BigDecimal.valueOf(1_000_000), 6, RoundingMode.HALF_UP);
        return in.add(out).add(cached);
    }
}
```

```java
// 本代码仅作学习材料参考
@Component
public class UsageAccumulator {

    private final StringRedisTemplate redis;

    // 每 session 的实时成本,Redis 累积
    public void record(UUID tenantId, UUID projectId, UUID userId, UUID sessionId, ModelUsage usage) {
        BigDecimal cost = usage.costUsd(prices);
        String dayKey = LocalDate.now().toString();

        redis.opsForHash().increment("cost:tenant:" + tenantId + ":" + dayKey, "total", cost.doubleValue());
        if (projectId != null) {
            redis.opsForHash().increment("cost:project:" + projectId + ":" + dayKey, "total", cost.doubleValue());
        }
        redis.opsForHash().increment("cost:user:" + userId + ":" + dayKey, "total", cost.doubleValue());
        redis.opsForHash().increment("cost:session:" + sessionId + ":" + dayKey, "total", cost.doubleValue());

        // 同时存 token 详情
        redis.opsForHash().increment("usage:tenant:" + tenantId + ":" + dayKey,
            usage.model() + ":in", usage.inputTokens());
        redis.opsForHash().increment("usage:tenant:" + tenantId + ":" + dayKey,
            usage.model() + ":out", usage.outputTokens());
    }

    public double sessionCost(UUID sessionId) {
        Object v = redis.opsForHash().get("cost:session:" + sessionId + ":" + LocalDate.now(), "total");
        return v == null ? 0 : Double.parseDouble(v.toString());
    }
}
```

```java
// 本代码仅作学习材料参考
@Component
public class BudgetGuard {

    public BudgetDecision check(UUID tenantId, UUID projectId, UUID userId, String requestedModel) {
        // 1. tenant monthly
        Budget tb = budgetRepo.findTenantMonthly(tenantId);
        double tSpent = accumulator.tenantMonthlySpent(tenantId);
        if (tb.hardLimit() != null && tSpent >= tb.hardLimit().doubleValue()) {
            return BudgetDecision.block("租户月度预算已达硬上限");
        }

        // 2. user daily
        Budget ub = budgetRepo.findUserDaily(userId);
        double uSpent = accumulator.userDailySpent(userId);
        if (ub.hardLimit() != null && uSpent >= ub.hardLimit().doubleValue()) {
            return BudgetDecision.block("用户日度预算已达硬上限");
        }

        // 3. soft limit:降级
        if (ub.softLimit() != null && uSpent >= ub.softLimit().doubleValue()) {
            return BudgetDecision.degrade(ub.degradeModel());
        }

        return BudgetDecision.allow(requestedModel);
    }
}
```

```java
// 本代码仅作学习材料参考
public Flux<Map<String, Object>> streamChat(Map<String, Object> req, TenantContext ctx) {
    String model = (String) req.get("model");

    BudgetDecision d = budgetGuard.check(ctx.tenantId(), ctx.projectId(), ctx.userId(), model);
    switch (d.action()) {
        case BLOCK -> {
            return Flux.error(new BudgetExceededException(d.reason()));
        }
        case DEGRADE -> {
            req.put("model", d.degradeModel());
            req.put("_degraded", true);
            // 通知前端
            eventBus.emit(AgentEvent.system(ctx.sessionId(), "model_degraded",
                Map.of("from", model, "to", d.degradeModel(), "reason", "budget")));
        }
    }

    return doRequest(req).doOnNext(chunk -> {
        // 解析 usage,实时累积
        if (chunk.containsKey("usage")) {
            ModelUsage u = parseUsage((Map<String, Object>) chunk.get("usage"), (String) req.get("model"));
            accumulator.record(ctx.tenantId(), ctx.projectId(), ctx.userId(), ctx.sessionId(), u);
        }
    });
}
```

```java
@Component
public class ModelRouter {

    public String route(String taskKind, TenantConfig config) {
        return switch (taskKind) {
            // 简单任务用便宜模型
            case "simple_qa", "summarize", "translate" -> pick(config.cheapModels());
            // 复杂编码用标准
            case "code_edit", "code_review", "analyze" -> pick(config.standardModels());

            default -> config.defaultModel();

    private String pick(List<String> candidates) {
        // 健康度优先 + 价格优先
        return candidates.stream()
            .filter(m -> healthChecker.isHealthy(m))
            .min(Comparator.comparingDouble(prices::inputPriceOf))
            .orElseThrow();
    }
}
```

```java
// 本代码仅作学习材料参考
@Component
public class BillingCycle {

    // 月初:生成发票
    @Scheduled(cron = "0 0 0 1 * *")
    public void generateInvoices() {
        for (Tenant t : tenantRepo.findByPlanNot("free")) {
            // 上月用量
            BigDecimal usage = usageService.lastMonthTotal(t.getId());
            BigDecimal baseFee = planService.baseFee(t.getPlan());
            BigDecimal seatsFee = planService.seatFee(t.getPlan(), t.getSeats());
            BigDecimal total = baseFee.add(seatsFee).add(usage);

            Invoice inv = new Invoice(t.getId(), total, ...);
            invoiceRepo.save(inv);

            if (t.getBillingMethod() == POSTPAID) {
                // 自动扣款
                stripeService.charge(inv);
            } else {
                // 发邮件通知付款
                mailService.sendInvoice(inv);
            }
        }
    }

    // 每小时:扣预付费余额
    @Scheduled(cron = "0 0 * * * *")
    public void deductCredits() {
        for (UUID tenantId : tenantRepo.findPrepaid()) {
            BigDecimal hourlyUsage = usageService.lastHour(tenantId);
            creditService.deduct(tenantId, hourlyUsage, "usage");

            // 余额低提醒
            BigDecimal balance = creditService.balance(tenantId);
            if (balance.compareTo(BigDecimal.valueOf(10)) < 0) {
                mailService.sendLowBalanceWarning(tenantId, balance);
            }
            if (balance.compareTo(BigDecimal.ZERO) <= 0) {
                // 余额耗尽,降级到免费层
                tenantService.downgradeToFree(tenantId);
            }
        }
    }
}
```

```java
@PostMapping("/api/billing/refund")
public void refund(@RequestBody RefundRequest req) {
    Payment p = paymentRepo.findById(req.paymentId()).orElseThrow();
    if (!"succeeded".equals(p.getStatus())) throw new IllegalStateException();

    // 1. 调 provider 退款
    Refund r = stripeService.refund(p.getProviderChargeId(), req.amount(), req.reason());

    // 2. 信用回滚
    creditService.refund(p.getTenantId(), req.amount(), "refund:" + r.getId());

    // 3. 标记
    p.setStatus("refunded");
    paymentRepo.save(p);

    // 4. 审计
    audit.emit("refund_processed", Map.of(
        "payment", p.getId(),
        "amount", req.amount(),
        "reason", req.reason(),
        "by", currentUser()
    ));
}
```

```java
public void reconcile() {
    // 1. 上游(Anthropic / OpenAI)账单 vs 内部统计
    BigDecimal upstream = upstreamBillingService.fetchYesterdayTotal();
    BigDecimal internal = usageService.yesterdayTotal();

    if (upstream.subtract(internal).abs().compareTo(BigDecimal.valueOf(1)) > 1) {
        alerting.notify("billing_mismatch",
            "upstream=" + upstream + ", internal=" + internal);
    }

    // 2. Stripe 收款 vs invoices
    for (Invoice inv : invoiceRepo.findYesterdayPaid()) {
        Payment p = stripeService.retrievePayment(inv);
        if (!inv.getAmount().equals(p.getAmount())) {
            alerting.notify("invoice_payment_mismatch", "invoice=" + inv.getNumber());
        }
    }
}
```

## 9. 预算预警

```java
// 本代码仅作学习材料参考
@Scheduled(cron = "0 */15 * * * *")  // 每 15 分钟检查
public void checkWarnings() {
    for (Tenant t : tenantRepo.findActive()) {
        double spent = accumulator.tenantMonthlySpent(t.getId());
        BigDecimal limit = budgetRepo.tenantHardLimit(t.getId());

        if (limit == null) continue;
        double pct = spent / limit.doubleValue();

        if (pct >= 1.0 && !notified(t.getId(), "100")) {
            notify(t, "预算已耗尽,服务降级");
            markNotified(t.getId(), "100");
        } else if (pct >= 0.8 && !notified(t.getId(), "80")) {
            notify(t, "本月已使用 80% 预算");
            markNotified(t.getId(), "80");
        } else if (pct >= 0.5 && !notified(t.getId(), "50")) {
            notify(t, "本月已使用 50% 预算");
            markNotified(t.getId(), "50");
        }
    }
}
```

## 11. 合规与发票

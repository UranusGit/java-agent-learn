### 2.1 State 对象

新建 `src/main/java/org/demo02/webclaude/agent/State.java`：

```java
// 本代码仅作学习材料参考
package org.demo02.webclaude.agent;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

public record State(
    UUID sessionId,
    List<Message> messages,
    String transition,
    long tokensIn,
    long tokensOut,
    boolean aborted
) {
    public static State initial(UUID sessionId, List<Message> history) {
        return new State(sessionId, new ArrayList<>(history), "init", 0, 0, false);
    }

    public State append(Message m) {
        var next = new ArrayList<>(messages);
        next.add(m);
        return new State(sessionId, next, transition, tokensIn, tokensOut, aborted);
    }

    public State withTransition(String t) {
        return new State(sessionId, messages, t, tokensIn, tokensOut, aborted);
    }

    public State addUsage(long in, long out) {
        return new State(sessionId, messages, transition, tokensIn + in, tokensOut + out, aborted);
    }

    public State markAborted() {
        return new State(sessionId, messages, transition, tokensIn, tokensOut, true);
    }
}
```

### 2.4 Agent Loop

新建 `src/main/java/org/demo02/webclaude/agent/AgentLoop.java`：

```java
// 本代码仅作学习材料参考
package org.demo02.webclaude.agent;

import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.messages.AssistantMessage;
import org.springframework.ai.chat.messages.UserMessage;
import org.springframework.ai.chat.model.ChatResponse;
import reactor.core.publisher.Flux;
import reactor.core.publisher.FluxSink;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

public class AgentLoop {

    private final ChatClient chatClient;

    public AgentLoop(ChatClient chatClient) {
        this.chatClient = chatClient;
    }

    public Flux<State> run(UUID sessionId, List<Message> history, String userInput) {
        return Flux.create(sink -> {
            // 1. 加入 user message
            UUID lastParentId = history.isEmpty() ? null : history.get(history.size() - 1).id();
            Message userMsg = Message.user(lastParentId, userInput);
            State state = State.initial(sessionId, history).append(userMsg).withTransition("thinking");
            sink.next(state);

            // 2. 调模型
            var msgs = new ArrayList<org.springframework.ai.chat.messages.Message>();
            history.forEach(m -> appendSpring(msgs, m));
            msgs.add(new UserMessage(userInput));

            chatClient.prompt()
                .messages(msgs)
                .stream()
                .chatResponse()
                .subscribe(
                    response -> onChunk(response, state, sink),
                    sink::error,
                    () -> onComplete(state, sink)
                );
        }, FluxSink.OverflowStrategy.BUFFER);
    }

    private void appendSpring(List<org.springframework.ai.chat.messages.Message> list, Message m) {
        if ("user".equals(m.role())) {
            m.content().stream()
                .filter(b -> b instanceof ContentBlock.Text)
                .forEach(b -> list.add(new UserMessage(((ContentBlock.Text) b).text())));
        } else if ("assistant".equals(m.role())) {
            m.content().stream()
                .filter(b -> b instanceof ContentBlock.Text)
                .forEach(b -> list.add(new AssistantMessage(((ContentBlock.Text) b).text())));
        }
    }

    private final StringBuilder buffer = new StringBuilder();

    private void onChunk(ChatResponse resp, State state, FluxSink<State> sink) {
        if (resp.getResult() == null || resp.getResult().getOutput() == null) return;
        String chunk = resp.getResult().getOutput().getText();
        if (chunk == null) return;
        buffer.append(chunk);
        sink.next(state.withTransition("streaming"));
    }

    private void onComplete(State state, FluxSink<State> sink) {
        Message assistantMsg = Message.assistant(null,
            List.of(new ContentBlock.Text(buffer.toString())));
        State finalState = state.append(assistantMsg).withTransition("end_turn");
        sink.next(finalState);
        sink.complete();
        buffer.setLength(0);
    }
}
```

> 注意：上面 `buffer` 是为了简化示例放在字段上，**实际不要这么写**（多并发请求会互相覆盖）。后续章节会改成 thread-safe 的 per-call builder。

### 3.1 改造 SessionWebSocketHandler

替换 `02-环境准备` 中的 `SessionWebSocketHandler`：

```java
// 本代码仅作学习材料参考
package org.demo02.webclaude.ws;

import com.fasterxml.jackson.databind.ObjectMapper;
import org.demo02.webclaude.agent.AgentLoop;
import org.demo02.webclaude.agent.ContentBlock;
import org.demo02.webclaude.agent.Message;
import org.demo02.webclaude.agent.State;
import org.demo02.webclaude.session.MessageEntity;
import org.demo02.webclaude.session.MessageRepository;
import org.demo02.webclaude.session.SessionRepository;
import org.springframework.stereotype.Component;
import org.springframework.web.socket.CloseStatus;
import org.springframework.web.socket.TextMessage;
import org.springframework.web.socket.WebSocketSession;
import org.springframework.web.socket.handler.TextWebSocketHandler;

import java.util.*;

@Component
public class SessionWebSocketHandler extends TextWebSocketHandler {

    private final AgentLoop agentLoop;
    private final SessionRepository sessionRepo;
    private final MessageRepository messageRepo;
    private final ObjectMapper om = new ObjectMapper();

    public SessionWebSocketHandler(AgentLoop agentLoop,
                                   SessionRepository sessionRepo,
                                   MessageRepository messageRepo) {
        this.agentLoop = agentLoop;
        this.sessionRepo = sessionRepo;
        this.messageRepo = messageRepo;
    }

    @Override
    protected void handleTextMessage(WebSocketSession ws, TextMessage raw) throws Exception {
        Map<?, ?> req = om.readValue(raw.getPayload(), Map.class);
        String type = (String) req.get("type");

        if ("user_input".equals(type)) {
            String sessionId = (String) req.get("session_id");
            String text = (String) req.get("text");
            UUID sid = UUID.fromString(sessionId);

            List<Message> history = loadHistory(sid);
            agentLoop.run(sid, history, text).subscribe(state -> {
                try {
                    ws.sendMessage(new TextMessage(om.writeValueAsString(toWire(state))));
                    if ("end_turn".equals(state.transition())) {
                        persistFinal(state);
                    }
                } catch (Exception e) {
                    e.printStackTrace();
                }
            });
        }
    }

    private Map<String, Object> toWire(State s) {
        Map<String, Object> wire = new HashMap<>();
        wire.put("type", "state");
        wire.put("session_id", s.sessionId().toString());
        wire.put("transition", s.transition());
        wire.put("messages", s.messages());
        return wire;
    }

    private List<Message> loadHistory(UUID sessionId) {
        // v1 简化：从 DB 加载，下一章改成 JSONL
        List<MessageEntity> entities = messageRepo.findBySessionIdOrdered(sessionId);
        // 这里做简化转换，省略；
        return List.of();
    }

    private void persistFinal(State s) {
        // v1 简化：只存最后一条 assistant 消息，下一章改成全量
        // 略
    }

    @Override
    public void afterConnectionClosed(WebSocketSession session, CloseStatus status) {}
}
```

> 注：这里故意把 `loadHistory` / `persistFinal` 留空 —— 在 07 章我们会完整实现 JSONL 持久化。本章只验证 Agent Loop 能流式返回。

### 4.2 Session Store

`src/store/sessionStore.ts`：

```typescript
// 本代码仅作学习材料参考
import { create } from 'zustand';
import type { Message } from '../types/message';

interface SessionState {
  messages: Message[];
  streaming: boolean;
  appendMessage: (m: Message) => void;
  updateLastAssistant: (text: string) => void;
  setStreaming: (s: boolean) => void;
  reset: () => void;
}

export const useSessionStore = create<SessionState>((set) => ({
  messages: [],
  streaming: false,
  appendMessage: (m) => set((s) => ({ messages: [...s.messages, m] })),
  updateLastAssistant: (text) =>
    set((s) => {
      const next = [...s.messages];
      for (let i = next.length - 1; i >= 0; i--) {
        if (next[i].role === 'assistant') {
          next[i] = { ...next[i], content: [{ type: 'text', text }] };
          break;
        }
      }
      return { messages: next };
    }),
  setStreaming: (b) => set({ streaming: b }),
  reset: () => set({ messages: [], streaming: false }),
}));
```

### 4.3 ChatPanel 组件

`src/components/ChatPanel.tsx`：

```tsx
// 本代码仅作学习材料参考
import { useState } from 'react';
import { SessionWS } from '../api/ws';
import { useSessionStore } from '../store/sessionStore';
import type { StateWire } from '../types/message';

const ws = new SessionWS('ws://localhost:8080/ws/session');
ws.connect();

ws.onMessage((m) => {
  if (m.type !== 'state') return;
  const wire = m as unknown as StateWire;
  // 简化处理：用 messages 数组刷新整个列表
  useSessionStore.setState({ messages: wire.messages });
  useSessionStore.getState().setStreaming(wire.transition !== 'end_turn');
});

export default function ChatPanel({ sessionId }: { sessionId: string }) {
  const [input, setInput] = useState('');
  const messages = useSessionStore((s) => s.messages);
  const streaming = useSessionStore((s) => s.streaming);

  const send = () => {
    if (!input.trim()) return;
    ws.send({ type: 'user_input', session_id: sessionId, text: input } as any);
    setInput('');
  };

  return (
    <div style={{ padding: 24, flex: 1 }}>
      <div style={{ marginBottom: 16 }}>
        {messages.map((m) => (
          <div key={m.id} style={{
            background: m.role === 'user' ? '#eef' : '#f5f5f5',
            padding: 8, margin: '4px 0', borderRadius: 4,
          }}>
            <strong>{m.role}</strong>
            <div>{m.content.map((c, i) =>
              c.type === 'text' ? <span key={i}>{c.text}</span> : null
            )}</div>
          </div>
        ))}
        {streaming && <div style={{ color: '#999' }}>... streaming</div>}
      </div>
      <textarea
        value={input}
        onChange={(e) => setInput(e.target.value)}
        rows={3}
        style={{ width: '100%' }}
        placeholder="输入消息..."
      />
      <button onClick={send} disabled={streaming}>发送</button>
    </div>
  );
}
```

# 31 - 多租户产品层:tenant、project、角色、邀请、SSO

> 本章目标:把 10 章的技术级多租户(JWT + tenant_id 隔离)升级为企业 SaaS 产品级。
> 完成后:租户自助注册 / 项目管理 / 角色权限 / 邀请流程 / SSO / 跨租户迁移 / 租户级独立配置。

> **关联章节**:
> - 10 章多租户技术隔离(本章基础);
> - 28 章限流 + 防滥用(本章注册流程);
> - 29 章计费(本章计费归属);

## 0. 为什么必须做

10 章的多租户只是技术隔离,真实场景:

| 场景 | 没做本章的痛苦 |
|------|---------------|
| 企业用户想自助注册账号 | 必须人工开通 |
| 团队 owner 想加 10 个成员 | 没有邀请流程 |
| 同一企业有 5 个项目,想分开管理 | 数据全混在一起 |
| 不同成员权限不同(viewer 只读 / admin 能改) | 全员 admin |
| 企业客户要求 SSO(Google / Azure AD) | 只能用密码登录 |
| 员工离职要交接项目 | 没有转让流程 |
| 企业要按 project 拆账 | 没有 project 概念 |

## 3. 注册与激活

### 3.1 注册流程

```java
// 本代码仅作学习材料参考
@PostMapping("/api/auth/register")
public RegisterResponse register(@RequestBody RegisterRequest req, HttpServletRequest http) {
    // 1. 限流 + 防爆破(28 章)
    rateLimiter.check("register:ip:" + ip(http), 5, Duration.ofHours(1));

    // 2. 唯一性检查
    if (userRepo.existsByEmail(req.email())) {
        throw new ConflictException("email_already_used");
    }

    // 3. 创建用户(tenant_id 还没有)
    User u = User.builder()
        .id(UUID.randomUUID())
        .email(req.email())
        .passwordHash(passwordEncoder.encode(req.password()))
        .status("pending_activation")
        .build();
    userRepo.save(u);

    // 4. 发激活邮件
    String token = secureToken();
    redis.opsForValue().set("activate:" + token, u.getId().toString(), Duration.ofHours(24));
    mailService.sendActivation(u.getEmail(), token);

    return new RegisterResponse(u.getId(), "请检查邮箱激活");
}

@GetMapping("/api/auth/activate")
public void activate(@RequestParam String token) {
    String userId = redis.opsForValue().get("activate:" + token);
    if (userId == null) throw new BadRequestException("激活链接无效或已过期");

    User u = userRepo.findById(UUID.fromString(userId)).orElseThrow();
    u.setStatus("active");
    u.setEmailVerifiedAt(Instant.now());
    userRepo.save(u);

    // 没租户的自动引导创建 / 加入
    response.sendRedirect("/onboarding");
}
```

```java
// 本代码仅作学习材料参考
@PostMapping("/api/tenants")
public Tenant createTenant(@RequestBody CreateTenantRequest req) {
    User u = currentUser();
    if (tenantMemberRepo.existsByUserId(u.getId())) {
        throw new ConflictException("每个账号只能属于一个租户");
    }

    // slug 唯一性
    if (tenantRepo.existsBySlug(req.slug())) {
        throw new ConflictException("slug 已被占用");
    }

    Tenant t = Tenant.builder()
        .id(UUID.randomUUID())
        .name(req.name())
        .slug(req.slug())
        .status("active")
        .plan("free")
        .ownerId(u.getId())
        .build();
    tenantRepo.save(t);

    // owner 自动加入
    tenantMemberRepo.save(new TenantMember(t.getId(), u.getId(), "owner",
        u.getId(), Instant.now(), Instant.now()));

    // 默认创建一个 "default" project
    projectService.createDefault(t.getId(), u.getId());

    // 同步 Stripe customer
    stripeService.createCustomer(t);

    return t;
}
```

```java
// 本代码仅作学习材料参考
@PostMapping("/api/tenants/{id}/invite")
public Invitation invite(@PathVariable UUID id, @RequestBody InviteRequest req) {
    TenantContext ctx = currentTenantContext();
    permissionChecker.requireTenantAdmin(ctx);

    // 检查 seat 配额
    long current = tenantMemberRepo.countByTenant(id);
    if (current >= planService.seatsLimit(ctx.plan())) {
        throw new ConflictException("已达 plan 上限,请升级");
    }

    String token = secureToken();
    Invitation inv = Invitation.builder()
        .id(UUID.randomUUID())
        .tenantId(id)
        .email(req.email())
        .role(req.role())
        .token(token)
        .invitedBy(ctx.userId())
        .expiresAt(Instant.now().plus(7, ChronoUnit.DAYS))
        .build();
    inviteRepo.save(inv);

    mailService.sendInvite(req.email(), ctx.tenantName(), token);
    return inv;
}

@PostMapping("/api/tenants/{id}/accept-invite")
public void acceptInvite(@PathVariable UUID id, @RequestBody AcceptInviteRequest req) {
    Invitation inv = inviteRepo.findByToken(req.token())
        .filter(i -> i.getExpiresAt().isAfter(Instant.now()))
        .filter(i -> !i.getTenantId().equals(id))
        .orElseThrow(() -> new NotFoundException("邀请无效"));

    User u = currentUser();

    // 已是该租户成员 → 提示
    if (tenantMemberRepo.existsByTenantAndUser(id, u.getId())) {
        throw new ConflictException("已是该租户成员");
    }

    tenantMemberRepo.save(new TenantMember(id, u.getId(), inv.getRole(),
        inv.getInvitedBy(), inv.getInvitedAt(), Instant.now()));
    inviteRepo.markAccepted(inv.getId());
}
```

### 4.3 移除 / 退出

```java
// 本代码仅作学习材料参考
@DeleteMapping("/api/tenants/{tid}/members/{uid}")
public void removeMember(@PathVariable UUID tid, @PathVariable UUID uid) {
    permissionChecker.requireTenantAdmin(currentTenantContext());

    TenantMember m = tenantMemberRepo.findById(new TenantMemberId(tid, uid))
        .orElseThrow(() -> new NotFoundException("成员不存在"));

    if ("owner".equals(m.getRole())) {
        throw new BadRequestException("不能移除 owner,需先转让");
    }

    // 同时从所有 project 移除
    projectMemberRepo.deleteByUser(uid);
    tenantMemberRepo.delete(m);

    audit.emit("member_removed", Map.of("tenant", tid, "user", uid, "by", currentUserId()));
}

@DeleteMapping("/api/tenants/{id}/members/me")
public void leaveTenant(@PathVariable UUID id) {
    TenantMember m = tenantMemberRepo.findById(new TenantMemberId(id, currentUserId()))
        .orElseThrow(NotFoundException::new);

    if ("owner".equals(m.getRole())) {
        throw new BadRequestException("owner 不能直接退出,需先转让");
    }

    // 释放资源(session / task owner 改成 admin 或归档)
    resourceService.reassignOrArchive(id, currentUserId());
    tenantMemberRepo.delete(m);
}
```

```java
// 本代码仅作学习材料参考
@PostMapping("/api/tenants/{id}/transfer-ownership")
public void transferOwnership(@PathVariable UUID id, @RequestBody TransferRequest req) {
    Tenant t = tenantRepo.findById(id).orElseThrow();
    if (!currentUserId().equals(t.getOwnerId())) {
        throw new ForbiddenException("仅 owner 可转让");
    }

    TenantMember newOwner = tenantMemberRepo.findById(new TenantMemberId(id, req.toUserId()))
        .orElseThrow(() -> new NotFoundException("目标成员不存在"));

    // 双重确认:输入 tenant slug 才能转让
    if (!req.confirmSlug().equals(t.getSlug())) {
        throw new BadRequestException("请输入正确的团队标识确认");
    }

    // 老的 owner 降级为 admin
    tenantMemberRepo.updateRole(id, t.getOwnerId(), "admin");
    // 新 owner 升级
    tenantMemberRepo.updateRole(id, req.toUserId(), "owner");
    tenantRepo.updateOwner(id, req.toUserId());

    audit.emit("ownership_transferred", Map.of(
        "tenant", id, "from", t.getOwnerId(), "to", req.toUserId()
    ));
}
```

```java
// 本代码仅作学习材料参考
@Component
public class PermissionChecker {

    public void requireTenantAdmin(TenantContext ctx) {
        if (!Set.of("owner", "admin").contains(ctx.tenantRole())) {
            throw new ForbiddenException("需要租户管理员权限");
        }
    }

    public void requireProjectAdmin(ProjectContext ctx) {
        if (!Set.of("owner", "admin").contains(ctx.tenantRole()) &&
            !"admin".equals(ctx.projectRole())) {
            throw new ForbiddenException("需要项目管理员权限");
        }
    }

    public void requireProjectMember(ProjectContext ctx) {
        if (ctx.projectRole() == null && !"viewer".equals(ctx.tenantRole())) {
            throw new ForbiddenException("需要项目成员权限");
        }
    }

    public boolean canEditProject(UUID projectId, UUID userId) {
        ProjectMember m = projectMemberRepo.findById(new ProjectMemberId(projectId, userId)).orElse(null);
        if (m != null && Set.of("admin", "member").contains(m.getRole())) return true;
        // 租户 owner / admin 也能编辑
        return isTenantAdmin(userId, currentTenantId());
    }
}
```

## 7. SSO 单点登录

### 7.1 OIDC(Google / Azure)

```java
// 本代码仅作学习材料参考
@RestController
@RequestMapping("/api/auth/sso")
public class SsoController {

    @GetMapping("/google/{tenantSlug}")
    public RedirectResult google(@PathVariable String tenantSlug) {
        SsoConnection conn = ssoService.find(tenantSlug, "google");
        String state = secureToken();
        redis.opsForValue().set("sso:state:" + state, tenantSlug, Duration.ofMinutes(10));

        String authUrl = "https://accounts.google.com/o/oauth2/v2/auth?" +
            "client_id=" + conn.config().get("client_id") +
            "&redirect_uri=" + config.getCallbackUrl("google") +
            "&response_type=code" +
            "&scope=openid email profile" +
            "&state=" + state;

        return new RedirectResult(authUrl);
    }

    @GetMapping("/callback/google")
    public RedirectResult callback(@RequestParam String code, @RequestParam String state) {
        String tenantSlug = redis.opsForValue().get("sso:state:" + state);
        if (tenantSlug == null) throw new BadRequestException("state 无效");

        // 1. 换 token
        GoogleTokenResponse token = googleClient.exchange(code);
        // 2. 拉 user info
        GoogleUserInfo info = googleClient.userInfo(token.accessToken());

        // 3. 匹配 / 创建用户
        ExternalIdentity ext = externalIdentityRepo.findByProviderAndExternalId("google", info.sub())
            .orElseGet(() -> provisionUser("google", info));

        // 4. 检查域名是否匹配
        SsoConnection conn = ssoService.find(tenantSlug, "google");
        String domain = info.email().split("@")[1];
        if (!Arrays.asList(conn.domains()).contains(domain)) {
            throw new ForbiddenException("邮箱域名不在白名单");
        }

        // 5. 签发 JWT
        String jwt = jwtService.issue(ext.getUserId(), tenantSlug);
        return new RedirectResult(config.getAppUrl() + "/sso-success?token=" + jwt);
    }
}
```

### 10.1 软删除

```java
// 本代码仅作学习材料参考
@DeleteMapping("/api/tenants/{id}")
public void deleteTenant(@PathVariable UUID id) {
    permissionChecker.requireTenantOwner(currentTenantContext());

    // 双重确认
    if (!req.confirmSlug().equals(tenant.getSlug())) {
        throw new BadRequestException();
    }

    // 软删除,30 天后硬删除
    tenant.setStatus("deleted");
    tenant.setDeletedAt(Instant.now());
    tenant.setScheduledPurgeAt(Instant.now().plus(30, ChronoUnit.DAYS));
    tenantRepo.save(tenant);

    // 立即停止所有活动
    sessionService.stopAll(id);
    taskService.cancelAll(id);

    mailService.sendDeletionNotice(tenant);
}

@Scheduled(cron = "0 0 3 * * *")
public void purgeDeleted() {
    for (Tenant t : tenantRepo.findScheduledPurge()) {
        hardDelete(t);
    }
}
```
